    task_id IDENTITY PRIMARY KEY,
    title VARCHAR(50) NOT NULL,
    description VARCHAR(1000)  NOT NULL,
    due_date DATE NOT NULL,
    priority ENUM('not that important', 'middle priority', 'very important') NOT NULL
);
